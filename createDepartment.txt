@startuml
skinparam sequenceArrowThickness 2
skinparam roundcorner 5
skinparam sequenceParticipant underline

actor "Usuario / Admin" as User

participant "DepartmentListComponent\n(Angular)" as UI
participant "DepartmentController\nPOST /departments" as Controller
participant "DepartmentService" as Service
participant "DepartmentRepository" as Repo
participant "CompanyRepository" as CompanyRepo
database "MongoDB\n(departments)" as DB

== Crear Departamento ==

User -> UI : Click "Nuevo departamento"
activate UI

UI -> UI : openCreate()\neditId.set(null)\nform = { companyId, name: '' }\nshowForm.set(true)
UI --> User : Muestra formulario modal

User -> UI : Ingresa companyId, name
User -> UI : Click "Guardar"

UI -> UI : save() → api.post('/departments', form)

UI -> Controller : POST /departments\n{companyId, name} + JWT
activate Controller

Controller -> Service : create(body)
activate Service

Service -> CompanyRepo : findById(companyId)
activate CompanyRepo
CompanyRepo -> DB : SELECT * FROM companies\nWHERE id=?
activate DB
DB --> CompanyRepo : Optional<Company>
deactivate DB

alt Company no encontrada
    CompanyRepo --> Service : Optional.empty()
    deactivate CompanyRepo
    Service --> Controller : 400 Bad Request\n"Empresa no encontrada"
    Controller --> UI : 400 Bad Request
    UI --> User : snack.open('Error', ...)
else Company existe
    CompanyRepo --> Service : Optional[Company]
    deactivate CompanyRepo

    Service -> Service : new Department()\n.setCompanyId(company.getId())\n.setName(name)

    Service -> Repo : save(department)
    activate Repo
    Repo -> DB : INSERT INTO departments\n{companyId, name, createdAt}
    activate DB
    DB --> Repo : Department (con id generado)
    deactivate DB
    Repo --> Service : Department
    deactivate Repo
    Service --> Controller : Department
    deactivate Service
    Controller --> UI : 201 Created – Department
    deactivate Controller
    UI -> UI : showForm.set(false)\nload()
    UI --> User : snack.open('Guardado', ...)
end

deactivate UI

@enduml
