@startuml
skinparam sequenceArrowThickness 2
skinparam roundcorner 5
skinparam sequenceParticipant underline

actor "Usuario / Admin" as User

participant "DepartmentListComponent\n(Angular)" as UI
participant "DepartmentController\nGET /departments" as Controller
participant "DepartmentService" as Service
participant "DepartmentRepository" as Repo
database "MongoDB\n(departments)" as DB

== Listar Departamentos ==

User -> UI : Navega a /departments
activate UI

UI -> UI : ngOnInit() → load()

UI -> Controller : GET /departments\n[?companyId] + JWT
activate Controller

Controller -> Service : findAll(companyId)
activate Service

alt companyId != null
    Service -> Repo : findByCompanyIdOrderByNameAsc(companyId)
else companyId == null
    Service -> Repo : findAll()
end

activate Repo
Repo -> DB : SELECT * FROM departments\n[WHERE companyId=?]
activate DB
DB --> Repo : List<Department>
deactivate DB
Repo --> Service : List<Department>
deactivate Repo
Service --> Controller : List<Department>
deactivate Service
Controller --> UI : 200 OK – List<Department>
deactivate Controller

UI -> UI : departments.set(v)\nloading.set(false)
UI --> User : Muestra tabla de departamentos
deactivate UI

@enduml
