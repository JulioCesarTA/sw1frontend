@startuml
skinparam sequenceArrowThickness 2
skinparam roundcorner 5
skinparam sequenceParticipant underline

actor "Usuario / Admin" as User

participant "DepartmentListComponent\n(Angular)" as UI
participant "DepartmentController\nPATCH /departments/{id}" as Controller
participant "DepartmentService" as Service
participant "DepartmentRepository" as Repo
participant "CompanyRepository" as CompanyRepo
database "MongoDB\n(departments)" as DB

== Actualizar Departamento ==

User -> UI : Click "Editar" en una fila
activate UI

UI -> UI : openEdit(department)\neditId.set(d.id)\nform = { companyId, name }\nshowForm.set(true)
UI --> User : Muestra formulario pre-cargado

User -> UI : Modifica name / companyId
User -> UI : Click "Guardar"

UI -> UI : save() → api.patch('/departments/{id}', form)

UI -> Controller : PATCH /departments/{id}\n{name?, companyId?} + JWT
activate Controller

Controller -> Service : update(id, body)
activate Service

Service -> Service : findOne(id)
Service -> Repo : findById(id)
activate Repo
Repo -> DB : SELECT * FROM departments\nWHERE id=?
activate DB
DB --> Repo : Optional<Department>
deactivate DB

alt Departamento no encontrado
    Repo --> Service : Optional.empty()
    deactivate Repo
    Service --> Controller : 404 Not Found\n"Departamento no encontrado"
    Controller --> UI : 404 Not Found
    UI --> User : snack.open('Error', ...)
else Departamento existe
    Repo --> Service : Optional[Department]
    deactivate Repo

    alt body contiene "companyId"
        Service -> CompanyRepo : findById(companyId)
        activate CompanyRepo
        CompanyRepo -> DB : SELECT * FROM companies\nWHERE id=?
        activate DB
        DB --> CompanyRepo : Optional<Company>
        deactivate DB

        alt Company no encontrada
            CompanyRepo --> Service : Optional.empty()
            deactivate CompanyRepo
            Service --> Controller : 400 Bad Request\n"Empresa no encontrada"
            Controller --> UI : 400 Bad Request
            UI --> User : snack.open('Error', ...)
        else Company existe
            CompanyRepo --> Service : Optional[Company]
            deactivate CompanyRepo
            Service -> Service : department.setCompanyId(companyId)
        end
    end

    Service -> Service : department.setName(name)

    Service -> Repo : save(department)
    activate Repo
    Repo -> DB : UPDATE departments\nSET name=?, companyId=?\nWHERE id=?
    activate DB
    DB --> Repo : Department actualizado
    deactivate DB
    Repo --> Service : Department
    deactivate Repo
    Service --> Controller : Department
    deactivate Service
    Controller --> UI : 200 OK – Department
    deactivate Controller
    UI -> UI : showForm.set(false)\nload()
    UI --> User : snack.open('Guardado', ...)
end

deactivate UI

@enduml
